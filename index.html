Create a file setup.json in the same directory.
Put an array of the currencies you own, and configure id fields according to CoinMarketCap API. 

{
  "localCurrency": {
    "id": "USD",
    "symbol": "$",
    "symbolPosition": "BEFORE"
  },
  "currenciesOwned": [
    {
      "id": "bitcoin",
      "amount": 0.04151
    },
    {
      "id": "ethereum",
      "amount": 1.194119638
    },
    {
      "id": "ripple",
      "amount": 1060.17
    },
    {
      "id": "stellar",
      "amount": 6629.75
    }
  ]
}
